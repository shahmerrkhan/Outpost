"use client";

import { motion } from "framer-motion";
import { ButtonHTMLAttributes } from "react";

export default function AnimatedButton(props: ButtonHTMLAttributes<HTMLButtonElement>) {
  return (
    <motion.button
      whileHover={{ scale: 1.03 }}
      whileTap={{ scale: 0.97 }}
      transition={{ duration: 0.15 }}
      {...(props as any)}
    />
  );
}
